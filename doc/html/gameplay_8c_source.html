105 int laby_gameplay_move_hero(Laby_Cell_str_t *src, Laby_Cell_str_t* dst) {

140 int laby_gameplay_move(Laby_Cell_str_t *src, Laby_Cell_str_t* dst) {
141  int (*ptrFunc)(Laby_Cell_str_t*,Laby_Cell_str_t*) = NULL; /* Pointeur vers la fonction de déplacement */
142 
143  if(src == NULL || dst == NULL)
144  return ERROR_PARAM_NULL;
145  if(src->entity == NULL)
146  return ERROR_PARAM_NULL;
147 
148  /* Table des déplacements. Une entité absente est une entité immobile. */
149  switch (src->entity->type) {
150  case LABY_HERO:
151  ptrFunc = laby_gameplay_move_hero;
152  break;
153  case LABY_ENEMY:
154  switch (src->entity->param1) {
155  case LABY_ENEMY_TYPE_BASIC:
156  ptrFunc = laby_gameplay_move_basic_enemy;
157  break;
158  case LABY_ENEMY_TYPE_GHOST:
159  ptrFunc = laby_gameplay_move_ghost;
160  break;
161  }
162  }
163 
164  if(ptrFunc == NULL)
165  return ERROR_INVALID_BEHAVIOR;
166  return ptrFunc(src,dst);
167 }
