315 int laby_game_enemies_move() {
316  int i,j;
317  Laby_Cell_str_t** cells;
318 
319 
320 
321  cells = (Laby_Cell_str_t**)lab.donnees;
322 
323  /* On réinitialise les indicateurs de mouvements : */
324  for(i = 0; i < lab.nblignes; i++)
325  for(j = 0; j <lab.nbcolonnes; j++){
326  if(cells[i][j].entity == NULL)
327  continue;
328  if(cells[i][j].entity->type != LABY_ENEMY)
329  continue;
330 
331  cells[i][j].entity->param2 = (cells[i][j].entity->param2 & !ENTITY_HAS_MOVED_FLAG);
332  }
333 
334 
335  for(i = 0; i < lab.nblignes; i++)
336  for(j = 0; j <lab.nbcolonnes; j++){
337  int random;
338  if(cells[i][j].entity == NULL)
339  continue;
340  if(cells[i][j].entity->type != LABY_ENEMY)
341  continue;
342 
343  /* Comme on parcourt les cases de manières descendantes, cette protection permet d'éviter de déplacer 2 fois la même entité. */
344  if(cells[i][j].entity->param2 & ENTITY_HAS_MOVED_FLAG){
345  continue;
346  }
347  cells[i][j].entity->param2 |= ENTITY_HAS_MOVED_FLAG;
348 
349  random = RANDOM(1,4);
350  switch(random){
351  case 1:
352  laby_gameplay_move_right(&cells[i][j]);
353  break;
354  case 2:
355  laby_gameplay_move_up(&cells[i][j]);
356  break;
357  case 3:
358  laby_gameplay_move_left(&cells[i][j]);
359  break;
360  case 4:
361  laby_gameplay_move_down(&cells[i][j]);
362  break;
363 
364 
365  }
366 
367  }
368 
369 
370 
371  return SUCCESS;
372 }

475 int laby_game_cycle(){
476  bool isNotDead = true;
477  bool Win = false;
478  Continue = true;
479 
480  /* Dans la boucle, les evenements à traiter sont :
481  - Dessiner le labyrinthe
482  - Verifier que le joueur n'a pas gagné
483  - Laisser le choix au joueur de se déplacer
484  - Vérifier que le joueur n'est pas mort
485  - Déplacer les monstres
486  - Vérifier qu'un monstre n'a pas tué le joueur
487  */
488 
489  laby_game_graphics_draw();
490  while(isNotDead && Continue) {
491  if(laby_game_check_victory()) {Continue = false; Win = true;}
492 
493  /* On vérifie que le joueur ne soit pas mort suite à un déplacement malencontreux */
494  isNotDead = (laby_game_player_move() != ERROR_HERO_DEAD);
495 
496  laby_game_enemies_move();
497 
498  /* On vérifie que le joueur ne soit pas mort suite à un déplacement ennemi */
499  isNotDead = !laby_game_check_if_player_dead() && isNotDead;
500  laby_game_graphics_draw();
501 
502  }
503 
504  if(!isNotDead) { /* Sortie par mort */
505  printf("\nPerdu ! :(\n");
506  usleep(3000000);
507  }
508  if(Win) { /* Victoire */
509  printf("\nBRAVO !\n");
510  usleep(2000000);
511  laby_game_scoring_handling();
512  }
513 
514 
515  return SUCCESS;
516 }
